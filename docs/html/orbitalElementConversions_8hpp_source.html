68template <typename Real, typename Vector6>
69Vector6 convertCartesianToKeplerianElements(
70 const Vector6& cartesianElements,
71 const Real gravitationalParameter,
72 const Real tolerance = 10.0 * std::numeric_limits<Real>::epsilon())
73{
74 assert(cartesianElements.size() == 6);
75
76 const Real pi = 3.14159265358979323846;
77
78 typedef std::vector<Real> Vector;
79 const Vector position = Vector({cartesianElements[0],
80 cartesianElements[1],
81 cartesianElements[2]});
82 const Real positionNormSquared = position[0] * position[0]
83 + position[1] * position[1]
84 + position[2] * position[2];
85 const Real positionNorm = std::sqrt(positionNormSquared);
86
87 const Vector velocity = Vector({cartesianElements[3],
88 cartesianElements[4],
89 cartesianElements[5]});
90 const Real velocityNormSquared = velocity[0] * velocity[0]
91 + velocity[1] * velocity[1]
92 + velocity[2] * velocity[2];
93
94 const Vector angularMomentum
95 = Vector({position[1] * velocity[2] - position[2] * velocity[1],
96 position[2] * velocity[0] - position[0] * velocity[2],
97 position[0] * velocity[1] - position[1] * velocity[0]});
98 const Real angularMomentumNormSquared = angularMomentum[0] * angularMomentum[0]
99 + angularMomentum[1] * angularMomentum[1]
100 + angularMomentum[2] * angularMomentum[2];
101 const Real angularMomentumNorm = std::sqrt(angularMomentumNormSquared);
102 const Vector angularMomentumUnitVector
103 = Vector({angularMomentum[0] / angularMomentumNorm,
104 angularMomentum[1] / angularMomentumNorm,
105 angularMomentum[2] / angularMomentumNorm});
106
107 const Real inclination = std::acos(angularMomentumUnitVector[2]);
108
109 const Real eccentricityPreMultiplier
110 = 1.0 / gravitationalParameter
111 * (velocityNormSquared - gravitationalParameter / positionNorm);
112
113 const Real positionDotVelocity = position[0] * velocity[0]
114 + position[1] * velocity[1]
115 + position[2] * velocity[2];
116
117 const Vector eccentricityVector
118 = Vector({eccentricityPreMultiplier * position[0]
119 - positionDotVelocity * velocity[0] / gravitationalParameter,
120 eccentricityPreMultiplier * position[1]
121 - positionDotVelocity * velocity[1] / gravitationalParameter,
122 eccentricityPreMultiplier * position[2]
123 - positionDotVelocity * velocity[2] / gravitationalParameter});
124
125 const Real eccentricityNormSquared = eccentricityVector[0] * eccentricityVector[0]
126 + eccentricityVector[1] * eccentricityVector[1]
127 + eccentricityVector[2] * eccentricityVector[2];
128
129 const Real eccentricity = std::sqrt(eccentricityNormSquared);
130
131 const Real semiMajorAxis
132 = (std::fabs(eccentricity - 1.0) < tolerance)
133 ? angularMomentumNormSquared / gravitationalParameter
134 : 1.0 / (2.0 / positionNorm - velocityNormSquared / gravitationalParameter);
135
136 const Vector ascendingNodeVector
137 = Vector({-angularMomentumUnitVector[1], angularMomentumUnitVector[0], 0.0});
138
139 const Real ascendingNodeVectorNormSquared
140 = ascendingNodeVector[0] * ascendingNodeVector[0]
141 + ascendingNodeVector[1] * ascendingNodeVector[1]
142 + ascendingNodeVector[2] * ascendingNodeVector[2];
143
144 const Real ascendingNodeVectorNorm = std::sqrt(ascendingNodeVectorNormSquared);
145
146 const Vector ascendingNodeUnitVector
147 = Vector({ascendingNodeVector[0] / ascendingNodeVectorNorm,
148 ascendingNodeVector[1] / ascendingNodeVectorNorm,
149 ascendingNodeVector[2] / ascendingNodeVectorNorm });
150
151 Real longitudeOfAscendingNode = 0.0;
152 if (std::fabs(inclination) < tolerance)
153 {
154 longitudeOfAscendingNode = std::numeric_limits<Real>::quiet_NaN();
155 }
156 else
157 {
158 const Real cosLongitudeOfAscendingNode = ascendingNodeUnitVector[0];
159 const Real sinLongitudeOfAscendingNode = ascendingNodeUnitVector[1];
160 const Real longitudeOfAscendingNodeShifted = std::atan2(sinLongitudeOfAscendingNode,
161 cosLongitudeOfAscendingNode);
162 longitudeOfAscendingNode = longitudeOfAscendingNodeShifted > 0.0
163 ? longitudeOfAscendingNodeShifted
164 : longitudeOfAscendingNodeShifted + 2.0 * pi;
165 }
166
167 Real argumentOfPeriapsis = 0.0;
168 if (std::fabs(eccentricity) < tolerance)
169 {
170 argumentOfPeriapsis = std::numeric_limits<Real>::quiet_NaN();
171 }
172 else
173 {
174 const Real cosArgumentOfPeriapsis
175 = ascendingNodeUnitVector[0] * eccentricityVector[0] / eccentricity
176 + ascendingNodeUnitVector[1] * eccentricityVector[1] / eccentricity
177 + ascendingNodeUnitVector[2] * eccentricityVector[2] / eccentricity;
178 const Real sinArgumentOfPeriapsis
179 = ((ascendingNodeUnitVector[1] * eccentricityVector[2]
180 - ascendingNodeUnitVector[2] * eccentricityVector[1]) / eccentricity)
181 * angularMomentumUnitVector[0]
182 + ((ascendingNodeUnitVector[2] * eccentricityVector[0]
183 - ascendingNodeUnitVector[0] * eccentricityVector[2]) / eccentricity)
184 * angularMomentumUnitVector[1]
185 + ((ascendingNodeUnitVector[0] * eccentricityVector[1]
186 - ascendingNodeUnitVector[1] * eccentricityVector[0]) / eccentricity)
187 * angularMomentumUnitVector[2];
188 const Real argumentOfPeriapsisShifted = std::atan2(sinArgumentOfPeriapsis,
189 cosArgumentOfPeriapsis);
190 argumentOfPeriapsis = argumentOfPeriapsisShifted > 0.0
191 ? argumentOfPeriapsisShifted
192 : argumentOfPeriapsisShifted + 2.0 * pi;
193 }
194
195 Real trueAnomaly = 0.0;
196 if (std::fabs(eccentricity) < tolerance)
197 {
198 if (std::fabs(inclination) < tolerance)
199 {
200 trueAnomaly = (velocity[0] < 0.0)
201 ? std::acos(position[0] / positionNorm)
202 : 2.0 * pi - std::acos(position[0] / positionNorm);
203 }
204 else
205 {
206 trueAnomaly = std::acos(position[0] / positionNorm * ascendingNodeUnitVector[0]
207 + position[1] / positionNorm * ascendingNodeUnitVector[1]
208 + position[2] / positionNorm * ascendingNodeUnitVector[2]);
209 }
210 }
211 else
212 {
213 const Real cosTrueAnomaly
214 = eccentricityVector[0] * position[0] / (eccentricity * positionNorm)
215 + eccentricityVector[1] * position[1] / (eccentricity * positionNorm)
216 + eccentricityVector[2] * position[2] / (eccentricity * positionNorm);
217 const Real sinTrueAnomaly
218 = ((eccentricityVector[1] * position[2]
219 - eccentricityVector[2] * position[1]) / (eccentricity * positionNorm))
220 * angularMomentumUnitVector[0]
221 + ((eccentricityVector[2] * position[0]
222 - eccentricityVector[0] * position[2]) / (eccentricity * positionNorm))
223 * angularMomentumUnitVector[1]
224 + ((eccentricityVector[0] * position[1]
225 - eccentricityVector[1] * position[0]) / (eccentricity * positionNorm))
226 * angularMomentumUnitVector[2];
227 const Real trueAnomalyShifted = std::atan2(sinTrueAnomaly, cosTrueAnomaly);
228 trueAnomaly = trueAnomalyShifted> 0.0 ? trueAnomalyShifted : trueAnomalyShifted + 2.0 * pi;
229 }
230
231 Vector6 keplerianElements = cartesianElements;
232 keplerianElements[0] = semiMajorAxis;
233 keplerianElements[1] = eccentricity;
234 keplerianElements[2] = inclination;
235 keplerianElements[3] = argumentOfPeriapsis;
236 keplerianElements[4] = longitudeOfAscendingNode;
237 keplerianElements[5] = trueAnomaly;
238 return keplerianElements;
239}

271template <typename Real, typename Vector6>
272Vector6 convertKeplerianToCartesianElements(
273 const Vector6& keplerianElements, const Real gravitationalParameter,
274 const Real tolerance = 10.0 * std::numeric_limits<Real>::epsilon())
275{
276 Vector6 cartesianElements = keplerianElements;
277
278 const Real semiMajorAxis = keplerianElements[semiMajorAxisIndex];
279 const Real eccentricity = keplerianElements[eccentricityIndex];
280 const Real inclination = keplerianElements[inclinationIndex];
281 const Real argumentOfPeriapsis = keplerianElements[argumentOfPeriapsisIndex];
282 const Real longitudeOfAscendingNode = keplerianElements[longitudeOfAscendingNodeIndex];
283 const Real trueAnomaly = keplerianElements[trueAnomalyIndex];
284
285 // Pre-compute sines and cosines of angles for efficient computation.
286 const Real cosineOfInclination = std::cos(inclination);
287 const Real sineOfInclination = std::sin(inclination);
288 const Real cosineOfArgumentOfPeriapsis = std::cos(argumentOfPeriapsis);
289 const Real sineOfArgumentOfPeriapsis = std::sin(argumentOfPeriapsis);
290 const Real cosineOfLongitudeOfAscendingNode = std::cos(longitudeOfAscendingNode);
291 const Real sineOfLongitudeOfAscendingNode = std::sin(longitudeOfAscendingNode);
292 const Real cosineOfTrueAnomaly = std::cos(trueAnomaly);
293 const Real sineOfTrueAnomaly = std::sin(trueAnomaly);
294
295 // Compute semi-latus rectum in the case the orbit is not a parabola.
296 Real semiLatusRectum = 0.0;
297 if (std::fabs(eccentricity - 1.0) > tolerance)
298 {
299 semiLatusRectum = semiMajorAxis * (1.0 - eccentricity * eccentricity);
300 }
301
302 // Else set the semi-latus rectum as the first element in the vector of Keplerian elements.
303 else
304 {
305 semiLatusRectum = keplerianElements[0];
306 }
307
308 // Compute the magnitude of the orbital radius, measured from the focal point.
309 const Real radiusMagnitude = semiLatusRectum / (1.0 + eccentricity * cosineOfTrueAnomaly);
310
311 // Define position and velocity in the perifocal coordinate system.
312 const Real xPositionPerifocal = radiusMagnitude * cosineOfTrueAnomaly;
313 const Real yPositionPerifocal = radiusMagnitude * sineOfTrueAnomaly;
314 const Real xVelocityPerifocal
315 = -std::sqrt(gravitationalParameter / semiLatusRectum) * sineOfTrueAnomaly;
316 const Real yVelocityPerifocal
317 = std::sqrt(gravitationalParameter / semiLatusRectum)
318 * (eccentricity + cosineOfTrueAnomaly);
319
320 // Compute scalar components of rotation matrix to rotate from periforcal to Earth-Centered
321 // Inertial (ECI) frame.
322 const Real rotationMatrixComponent11
323 = (cosineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis
324 -sineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis * cosineOfInclination);
325 const Real rotationMatrixComponent12
326 = (-cosineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis
327 -sineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis * cosineOfInclination);
328
329 const Real rotationMatrixComponent21
330 = (sineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis
331 + cosineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis * cosineOfInclination);
332 const Real rotationMatrixComponent22
333 = (-sineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis
334 + cosineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis * cosineOfInclination);
335
336 const Real rotationMatrixComponent31 = (sineOfArgumentOfPeriapsis * sineOfInclination);
337 const Real rotationMatrixComponent32 = (cosineOfArgumentOfPeriapsis * sineOfInclination);
338
339 // Compute Cartesian position and velocities.
340 cartesianElements[xPositionIndex] = rotationMatrixComponent11 * xPositionPerifocal
341 + rotationMatrixComponent12 * yPositionPerifocal;
342
343 cartesianElements[yPositionIndex] = rotationMatrixComponent21 * xPositionPerifocal
344 + rotationMatrixComponent22 * yPositionPerifocal;
345
346 cartesianElements[zPositionIndex] = rotationMatrixComponent31 * xPositionPerifocal
347 + rotationMatrixComponent32 * yPositionPerifocal;
348
349 cartesianElements[xVelocityIndex] = rotationMatrixComponent11 * xVelocityPerifocal
350 + rotationMatrixComponent12 * yVelocityPerifocal;
351
352 cartesianElements[yVelocityIndex] = rotationMatrixComponent21 * xVelocityPerifocal
353 + rotationMatrixComponent22 * yVelocityPerifocal;
354
355 cartesianElements[zVelocityIndex] = rotationMatrixComponent31 * xVelocityPerifocal
356 + rotationMatrixComponent32 * yVelocityPerifocal;
357
358 return cartesianElements;
359}

375template <typename Real>
376Real convertTrueAnomalyToEllipticalEccentricAnomaly(const Real trueAnomaly,
377 const Real eccentricity)
378{
379 assert(eccentricity >= 0.0 && eccentricity < 1.0);
380
381 // Compute sine and cosine of eccentric anomaly.
382 const Real sineOfEccentricAnomaly
383 = std::sqrt(1.0 - std::pow(eccentricity, 2.0))
384 * std::sin(trueAnomaly) / (1.0 + eccentricity * std::cos(trueAnomaly));
385 const Real cosineOfEccentricAnomaly
386 = (eccentricity + std::cos(trueAnomaly))
387 / (1.0 + eccentricity * std::cos(trueAnomaly));
388
389 // Return elliptical eccentric anomaly.
390 return std::atan2(sineOfEccentricAnomaly, cosineOfEccentricAnomaly);
391}

408template <typename Real>
409Real convertTrueAnomalyToHyperbolicEccentricAnomaly(const Real trueAnomaly,
410 const Real eccentricity)
411{
412 assert(eccentricity > 1.0);
413
414 // Compute hyperbolic sine and hyperbolic cosine of hyperbolic eccentric anomaly.
415 const Real hyperbolicSineOfHyperbolicEccentricAnomaly
416 = std::sqrt(std::pow(eccentricity, 2.0) - 1.0)
417 * std::sin(trueAnomaly) / (1.0 + std::cos(trueAnomaly));
418
419 const Real hyperbolicCosineOfHyperbolicEccentricAnomaly
420 = (std::cos(trueAnomaly) + eccentricity) / (1.0 + std::cos(trueAnomaly));
421
422 // Return hyperbolic eccentric anomaly.
423 // The inverse hyperbolic tangent is computed here manually, since the atanh() function is not
424 // available in older C++ compilers.
425 const Real angle
426 = hyperbolicSineOfHyperbolicEccentricAnomaly
427 / hyperbolicCosineOfHyperbolicEccentricAnomaly;
428 return 0.5 * (std::log(1.0 + angle) - std::log(1.0 - angle));
429}

450template <typename Real>
451Real convertTrueAnomalyToEccentricAnomaly(const Real trueAnomaly, const Real eccentricity)
452{
453 assert(eccentricity >= 0.0
454 && (std::fabs(eccentricity - 1.0) > std::numeric_limits<Real>::epsilon()));
455
456 Real eccentricAnomaly = 0.0;
457
458 // Check if orbit is elliptical and compute eccentric anomaly.
459 if (eccentricity >= 0.0 && eccentricity < 1.0)
460 {
461 eccentricAnomaly
462 = convertTrueAnomalyToEllipticalEccentricAnomaly(trueAnomaly, eccentricity);
463 }
464
465 // Check if orbit is hyperbolic and compute eccentric anomaly.
466 else if (eccentricity > 1.0)
467 {
468 eccentricAnomaly
469 = convertTrueAnomalyToHyperbolicEccentricAnomaly(trueAnomaly, eccentricity);
470 }
471
472 return eccentricAnomaly;
473}

577template <typename Real>
578Real convertEllipticalEccentricAnomalyToTrueAnomaly(const Real ellipticEccentricAnomaly,
579 const Real eccentricity)
580{
581 const Real sineOfTrueAnomaly = std::sqrt(1.0 - eccentricity * eccentricity)
582 * std::sin(ellipticEccentricAnomaly)
583 / (1.0 - eccentricity * std::cos(ellipticEccentricAnomaly));
584
585 const Real cosineOfTrueAnomaly
586 = (std::cos(ellipticEccentricAnomaly) - eccentricity)
587 / (1.0 - eccentricity * std::cos(ellipticEccentricAnomaly));
588
589 return std::atan2(sineOfTrueAnomaly, cosineOfTrueAnomaly);
590}

602template <typename Real>
603Real convertHyperbolicEccentricAnomalyToTrueAnomaly(const Real hyperbolicEccentricAnomaly,
604 const Real eccentricity)
605{
606 const Real sineOfTrueAnomaly
607 = std::sqrt(eccentricity * eccentricity - 1.0)
608 * std::sinh(hyperbolicEccentricAnomaly)
609 / (eccentricity * std::cosh(hyperbolicEccentricAnomaly) - 1.0);
610
611 const Real cosineOfTrueAnomaly
612 = (eccentricity - std::cosh(hyperbolicEccentricAnomaly))
613 / (eccentricity * std::cosh(hyperbolicEccentricAnomaly) - 1.0);
614
615 return std::atan2(sineOfTrueAnomaly, cosineOfTrueAnomaly);
616}

739template <typename Real, typename Integer>
740Real convertEllipticalMeanAnomalyToEccentricAnomaly(
741 const Real eccentricity,
742 const Real meanAnomaly,
743 const Real rootFindingTolerance = 1.0e-3 * std::numeric_limits<Real>::epsilon(),
744 const Integer maximumIterations = 100)
745{
746 assert(eccentricity >= 0.0 && eccentricity < (1.0 - 1.0e-11));
747
748 const Real pi = 3.14159265358979323846;
749
750 // Set mean anomaly to domain between 0 and 2pi.
751 Real meanAnomalyShifted = std::fmod(meanAnomaly, 2.0 * pi);
752 if (meanAnomalyShifted <0.0)
753 {
754 meanAnomalyShifted += 2.0 * pi;
755 }
756
757 Real eccentricAnomaly = std::numeric_limits<Real>::quiet_NaN();
758
759 // Set the initial guess for the eccentric anomaly.
760 // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
761 // If this scheme is changed, please run a very extensive test suite. The Newton-Raphson
762 // root-finding scheme tends to be chaotic for very specific combinations of mean anomaly
763 // and eccentricity. Various random tests of 100,000,000 samples were done to verify the
764 // default scheme for the initial guess. (Musegaas, 2012).
765 Real initialGuess = 0.0;
766 if (meanAnomalyShifted > pi)
767 {
768 initialGuess = meanAnomalyShifted - eccentricity;
769 }
770 else
771 {
772 initialGuess = meanAnomalyShifted + eccentricity;
773 }
774
775 // Execute Newton-Raphson root-finding algorithm.
776 eccentricAnomaly = initialGuess;
777 for (int i = 0; i < maximumIterations + 1; ++i)
778 {
779 if (i == maximumIterations)
780 {
781 throw std::runtime_error(
782 "ERROR: Maximum iterations for Newton-Raphson root-finding exceeded!");
783 }
784
785 const Real nextEccentricAnomaly
786 = eccentricAnomaly
787 - computeEllipticalKeplerFunction(eccentricAnomaly,
788 eccentricity,
789 meanAnomalyShifted)
790 / computeFirstDerivativeEllipticalKeplerFunction(eccentricAnomaly,
791 eccentricity);
792
793 const Real eccentricAnomalyDifference = eccentricAnomaly - nextEccentricAnomaly;
794
795 eccentricAnomaly = nextEccentricAnomaly;
796
797 if (eccentricAnomalyDifference < rootFindingTolerance)
798 {
799 break;
800 }
801 }
802
803 // Return eccentric anomaly.
804 return eccentricAnomaly;
805}
